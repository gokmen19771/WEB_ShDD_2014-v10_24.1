SELECT        dbo.FORM_YATAYDATA.FormAdi, dbo.Kullanicilar.KullaniciAdSoyad, dbo.KURUM_LISTESI.Kurum AS DenetleyeceðiKurum, 
                         dbo.FORM_YATAYDATA.IlKod AS DenetlediðiÝl, dbo.Kullanicilar.Adres, dbo.Kullanicilar.Tel, KURUM_LISTESI_1.Kurum AS KendiKurumu, 
                         COUNT(dbo.FORM_YATAYDATA.DenekAd) AS HastaSayýsý, dbo.Kullanicilar.Unvan, dbo.Kullanicilar.Brans, dbo.Kullanicilar.IlgiAlani, dbo.Kullanicilar.IlKod, 
                         dbo.Kullanicilar.KurumKod

                         dbo.Kullanicilar ON dbo.FORM_YATAYDATA.AnketorId = dbo.Kullanicilar.KullaniciID INNER JOIN
                         dbo.KURUM_LISTESI ON dbo.FORM_YATAYDATA.KurumKod = dbo.KURUM_LISTESI.KurumKod LEFT OUTER JOIN
                         dbo.KURUM_LISTESI AS KURUM_LISTESI_1 ON dbo.Kullanicilar.KurumKod = KURUM_LISTESI_1.KurumKod
GROUP BY dbo.FORM_YATAYDATA.FormAdi, dbo.Kullanicilar.KullaniciAdSoyad, dbo.Kullanicilar.Adres, dbo.Kullanicilar.Tel, dbo.KURUM_LISTESI.Kurum,

                         dbo.Kullanicilar.KurumKod
